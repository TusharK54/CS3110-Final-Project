Module Minesweeper

module Minesweeper: sig .. end
type result =
| Win
| Lose
| Incomplete 
type appearance =
| Hidden
| Uncovered
| Flag 
type contents =
| Number of int
| Mine 
type box = appearance * contents 
type position = int * int 
type state 
exception Invalid_pos
val dimensions : int * int

dimensions is the dimensions of the grid.

val mines : int

mines is the number of mines.

val new_game : state

new_game is a randomly initialized state of minesweeper.

val get_box : state -> position -> box

get_box t pos is the box located at position pos in state t. Raises Invalid_Pos if pos is not a valid position.

val set_box : state ->
       position -> box -> state

set_box t pos b is the state t with the box located at position pos set to b. Raises Invalid_Pos if pos is not a valid position.

val game_result : state -> result

game_result t is Win if the player has won, Lose if the player has lost, and Incomplete if the game is not finished.

val flag : state -> position -> state

flag t pos is the state t with the box at position pos marked as a flag if it has not yet been uncovered.

val uncover : state -> position -> state

uncover t pos is the state after uncovering the box at position pos.
